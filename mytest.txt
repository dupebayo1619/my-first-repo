#!/bin/bash

# Define your tools
tools=("shellcheck" "net-tools")

echo "--- Starting System Audit ---"

for tool in "${tools[@]}"; do
    if command -v "$tool" > /dev/null; then
        echo "[OK] $tool is already installed."
        
        # Logic: Print the version for the user
        if [[ "$tool" == "shellcheck" ]]; then
            shellcheck --version | grep "version:"
        elif [[ "$tool" == "net-tools" ]]; then
            # net-tools doesn't have a simple --version, so we check ifconfig
            ifconfig --version | head -n 1
        fi
    else
        echo "[MISSING] $tool not found. Installing..."
        sudo apt update -qq && sudo apt install -y "$tool"
        echo "[FIXED] $tool has been installed."
    fi
    echo "----------------------------"
done

echo "--- Audit Complete ---"
